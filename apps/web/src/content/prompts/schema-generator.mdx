  const faqCollection = defineCollection({
    loader: glob({ pattern: '**/*.mdx', base: './src/content/faq' }),
    schema: z.object({
      title_fr: z.string().min(1),
      title_en: z.string().min(1),
      slug: z.string().regex(/^[a-z0-9-]+$/),
